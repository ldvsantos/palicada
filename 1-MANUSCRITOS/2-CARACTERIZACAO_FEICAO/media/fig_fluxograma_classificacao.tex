% 8) Forma da seção transversal

% --------------------------------------------------

% feições -> profundidade

\draw (hubA) |- (prof1.north);
\draw (hubA) |- (prof2.north);

\draw (prof2.south) -- (vocoroca.north);

\draw (hubFam) |- (relevo.north);
